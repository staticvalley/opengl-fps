cmake_minimum_required(VERSION 3.20)
project(OpenGLFPS LANGUAGES C CXX)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)

# fetch sdl3
find_package(SDL3 CONFIG REQUIRED)

# fetch opengl
find_package(OpenGL REQUIRED)

#fetch doxygen
find_package(Doxygen)

if(DOXYGEN_FOUND)
    set(DOXYGEN_PROJECT_NAME "${PROJECT_NAME}")
    set(DOXYGEN_OUTPUT_DIRECTORY ${CMAKE_SOURCE_DIR}/docs)
    set(DOXYGEN_RECURSIVE YES)
    set(DOXYGEN_EXTRACT_ALL YES)
    set(DOXYGEN_GENERATE_LATEX NO)

    doxygen_add_docs(
        docs
        ${CMAKE_SOURCE_DIR}/src
        ${CMAKE_SOURCE_DIR}/include
        COMMENT "creating documentation..."
    )
endif()

# add glad
add_library(glad libs/glad/src/glad.c)

target_include_directories(glad PUBLIC libs/glad/include)

# add Dear imgui
add_library(imgui
    libs/imgui/imgui.cpp
    libs/imgui/imgui_draw.cpp
    libs/imgui/imgui_tables.cpp
    libs/imgui/imgui_widgets.cpp
    libs/imgui/imgui_demo.cpp
    libs/imgui/backends/imgui_impl_sdl3.cpp
    libs/imgui/backends/imgui_impl_opengl3.cpp
)

target_include_directories(imgui PUBLIC libs/imgui libs/imgui/backends)

# imgui needs sdl3 for binding
target_link_libraries(imgui PRIVATE SDL3::SDL3)

add_executable(${PROJECT_NAME}
    src/main.cpp
    src/rendering/Shader.cpp
    src/rendering/Mesh.cpp
    src/rendering/Model.cpp
    src/rendering/Texture.cpp
    src/rendering/Material.cpp
    src/rendering/Skybox.cpp
    src/scene/Camera.cpp
    src/game/Player.cpp
    src/physics/Collision.cpp
    src/physics/PlayerMovement.cpp
    src/debug/DebugDrawer.cpp
    src/mapping/WADFile.cpp
    src/mapping/BSPFile.cpp
    src/mapping/BSPMap.cpp
)

target_include_directories(${PROJECT_NAME} PRIVATE
    src/
    include/
    include/rendering/
    include/scene/
    include/physics/
    include/game/
    include/debug/
    include/mapping/
    libs/glad/include
    libs/stb
    libs/glm
    libs/tinyobj
)

target_link_libraries(${PROJECT_NAME} PRIVATE
    SDL3::SDL3
    OpenGL::GL
    glad
    imgui
)

add_custom_target(copy_assets ALL
    COMMAND ${CMAKE_COMMAND} -E copy_directory
        ${CMAKE_SOURCE_DIR}/assets
        $<TARGET_FILE_DIR:${PROJECT_NAME}>/assets
)

add_dependencies(${PROJECT_NAME} copy_assets)

# copy sdl3 dll to output dir on windows
if(WIN32)
    add_custom_command(TARGET ${PROJECT_NAME} POST_BUILD
        COMMAND ${CMAKE_COMMAND} -E copy_if_different
            $<TARGET_FILE:SDL3::SDL3>
            $<TARGET_FILE_DIR:${PROJECT_NAME}>
    )
endif()

# turn on compiler warnings
if(MSVC)
    target_compile_options(${PROJECT_NAME} PRIVATE /W4)
else()
    target_compile_options(${PROJECT_NAME} PRIVATE -Wall -Wextra -Wpedantic)
endif()
